assume cs:code


data segment
    db 10 dup(0)
data ends

code segment
start:   mov ax,12666
           mov bx,data
           mov ds,bx
           mov si,0
           call dtoc
           mov dh,8
           mov dl,3
           mov cl,2
           call show_str

dtoc:
        mov dx,0
        mov bx,10
        mov si,0
        div bx
        push dx
        inc si
        mov cx,ax
        jcxz ok
       jmp short dtoc

ok:   mov cx,si
        mov si,0

s:      pop bx
         add bx,30H
         mov ds:[si],bx
         inc si
         loop s
         mov [si], cl
         mov si,0
         ret
        

show_str:
         push ax
         push bx
         push cx
         push dx
         push es
         push si
         push di
         mov ax,0B800h
         mov es,ax
         mov al,160
         mul dh
         mov bx,ax
         mov al,2
         mul dl
         add bx,ax
         
         mov al,cl

help: 
         mov cl,ds:[si]
         jcxz exit
         mov es:[bx+di],cl
         mov es:[bx+di+1],al
         inc si
         add di,2
         jmp short help

exit:
         pop di
         pop si
         pop es
         pop dx
         pop cx
         pop bx
         pop ax
        ret
code ends
end start
